    \zihao{7}
    \caption{\kaishu \zihao{-5}适用于P256曲线的逐字蒙哥马利乘法(P256-WW-MM)}
    \label{alg1}
    \LinesNumbered %要求显示行号
    \KwIn{素数 $p<2^{256}, 0\leq a,b<p$}  %输入参数
    \KwOut{$a \times b \times 2^{-256} \bmod p$}
    \kwInit{$p_{0}=-p^{-1} \bmod 2^{64}$}
    \emph {$T=a \times b$}\;
    \For{$\mathrm{i \; from \; 0 \; by \; 1 \; to \; 4}$}{
        $T_1 = T \bmod 2^{64}$\;
        $Y = T_1 \times p_{0} \bmod 2^{64}$\;
        $T_2 =Y \times p$\;
        $T_3 =(T+T_2)$\;
        $T=T_3 / 2^{64}$
    }
    \uIf{$T \geq p$}{
        $X=T-p$ \;
    }
    \Else{
        $X=T$
    }
    \emph {return $X$}\;
\end{algorithm}
